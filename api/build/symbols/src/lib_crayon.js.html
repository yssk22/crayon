1 /**
  2  * @namespace Holds core static functions
  3  * @class Crayon class
  4  * @description
  5  * This class is originated from jQuery.core functions and provides basic functions for Crayon libraries.
  6  * @see <a href="http://jqueryjs.googlecode.com/svn/tags/1.3.2/src/core.js">jQuery 1.3.2</a>
  7  */
  8 var Crayon = {};
  9 
 10 /**
 11  * @function
 12  * @return true when object is Function, otherwise false.
 13  */
 14 Crayon.isFunction = function( obj ) {
 15   return toString.call(obj) === "[object Function]";
 16 };
 17 
 18 /**
 19  * @function
 20  * @return true when object is Array, otherwise false.
 21  */
 22 Crayon.isArray = function( obj ) {
 23   return toString.call(obj) === "[object Array]";
 24 };
 25 
 26 /**
 27  * @function
 28  * @return true when object is Object, otherwise false.
 29  */
 30 Crayon.isObject = function(obj){
 31   return toString.call(obj) === "[object Object]";
 32 };
 33 
 34 /**
 35  * Extends a objecct or Crayon itself.
 36  * @param [Object] object
 37  */
 38 Crayon.extend = function(){
 39   // copy reference to target object
 40   var target = arguments[0] || {}, i = 1, length = arguments.length, deep = false, options, name, src, copy;
 41 
 42   // Handle a deep copy situation
 43   if ( typeof target === "boolean" ) {
 44     deep = target;
 45     target = arguments[1] || {};
 46     // skip the boolean and the target
 47     i = 2;
 48   }
 49 
 50   // Handle case when target is a string or something (possible in deep copy)
 51   if ( typeof target !== "object" && !Crayon.isFunction(target) ) {
 52     target = {};
 53   }
 54 
 55   // extend Crayon itself if only one argument is passed
 56   if ( length === i ) {
 57     target = this;
 58     --i;
 59   }
 60 
 61   for ( ; i < length; i++ ) {
 62     // Only deal with non-null/undefined values
 63     if ( (options = arguments[ i ]) != null ) {
 64       // Extend the base object
 65       for ( name in options ) {
 66         src = target[ name ];
 67         copy = options[ name ];
 68 
 69         // Prevent never-ending loop
 70         if ( target === copy ) {
 71           continue;
 72         }
 73 
 74         // Recurse if we're merging object values
 75         if ( deep && copy && typeof copy === "object" && !copy.nodeType ) {
 76           var clone;
 77 
 78           if ( src ) {
 79             clone = src;
 80           } else if ( Crayon.isArray(copy) ) {
 81             clone = [];
 82           } else if ( Crayon.isObject(copy) ) {
 83             clone = {};
 84           } else {
 85             clone = copy;
 86           }
 87 
 88           // Never move original objects, clone them
 89           target[ name ] = Crayon.extend( deep, clone, copy );
 90 
 91           // Don't bring in undefined values
 92         } else if ( copy !== undefined ) {
 93           target[ name ] = copy;
 94         }
 95       }
 96     }
 97   }
 98 
 99   // Return the modified object
100   return target;
101 };
