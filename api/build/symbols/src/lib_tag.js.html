1 /**
  2  * @class
  3  * @namespace Holds functionality for escaping charactors.
  4  * @requires crayon/lib/core.js
  5  * @requires crayon/lib/escape.js
  6  */
  7 Crayon.Tag = {
  8   /**
  9    * Retrurns a tag string.
 10    * @param {String} name a tag name such as "p".
 11    * @param [Object] options kev-value pairs for the tag attributes. The default is <tt>{}</tt>.
 12    * @param [Boolean] open if true, the closer is excluded. The default is fasle.
 13    * @param [Boolean] escape if true, the special chars in tag attributes specified by <tt>options</tt> is escaped.
 14    */
 15   tag : function(name, options, open, escape){
 16     if( escape === undefined ){
 17       escape = true;
 18     }
 19     var str = "<" + name;
 20     if(options){
 21       for(var attr_name in options){
 22         str += " " + attr_name + '="' + (escape ? h(options[attr_name]) : options[attr_name]) + '"';
 23       }
 24     }
 25     return str + (open ? ">" : "/>");
 26   },
 27 
 28   /**
 29    * Retrurns a tag string.
 30    * @param {String} name a tag name such as "p".
 31    * @param {String or Function} content The text inserted into the tag. If a function is passed, the returned value is inserted.
 32    * @param [Object] options kev-value pairs for the tag attributes. The default is <tt>{}</tt>.
 33    * @param [Boolean] escape if true, the special chars in tag attributes specified by <tt>options</tt> is escaped.
 34    */
 35   content_tag : function(name, content, options, escape){
 36     var str = tag(name, options, true, escape);
 37     str += Crayon.isFunction(content) ? content() : content.toString();
 38     return str + "</" + name + ">";
 39   }
 40 };
 41 
 42 if( !this.do_not_import_global ){
 43   Crayon.extend(this, Crayon.Tag);
 44 }else{
 45   Crayon.extend(Crayon.Tag);
